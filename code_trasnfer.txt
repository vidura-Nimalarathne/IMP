#!/usr/bin/env python
import rospy
from std_msgs.msg import Bool, String
from geometry_msgs.msg import Twist

class SimpleSearch(object):
    def __init__(self):
        rospy.init_node('simple_autonomous_search')

        self.cmd_pub = rospy.Publisher('/cmd_vel', Twist, queue_size=1)
        self.gui_pub = rospy.Publisher('/gui_status', String, queue_size=1)

        rospy.Subscriber('/dog_detected', Bool, self.dog_callback)
        rospy.Subscriber('/start_autonomous', Bool, self.start_callback)

        self.dog_found = False
        self.active = False

    def dog_callback(self, msg):
        if msg.data:
            rospy.logwarn("DOG FOUND!")
            self.dog_found = True
            self.stop_robot()
            self.gui_pub.publish("DOG FOUND")
            self.active = False

    def start_callback(self, msg):
        if msg.data:
            rospy.loginfo("Autonomous search started")
            self.active = True
            self.dog_found = False
            self.run_sequence()

    def stop_robot(self):
        self.cmd_pub.publish(Twist())

    def move_forward(self, duration):
        twist = Twist()
        twist.linear.x = 0.15
        rate = rospy.Rate(10)
        start = rospy.Time.now()

        while (rospy.Time.now() - start).to_sec() < duration:
            if self.dog_found or not self.active:
                return
            self.cmd_pub.publish(twist)
            rate.sleep()

        self.stop_robot()

    def turn_right_90(self):
        twist = Twist()
        twist.angular.z = -0.5   # CW turn
        duration = 2.5           # adjust if needed

        rate = rospy.Rate(10)
        start = rospy.Time.now()

        while (rospy.Time.now() - start).to_sec() < duration:
            if self.dog_found or not self.active:
                return
            self.cmd_pub.publish(twist)
            rate.sleep()

        self.stop_robot()

    def run_sequence(self):
        rospy.loginfo("Step 1: Forward 3 sec")
        self.move_forward(3.0)

        if self.dog_found: return

        rospy.sleep(1.0)

        rospy.loginfo("Step 2: Turn 90 CW")
        self.turn_right_90()

        if self.dog_found: return

        rospy.sleep(1.0)

        rospy.loginfo("Step 3: Forward 2 sec")
        self.move_forward(2.0)

        self.stop_robot()
        self.active = False
        rospy.loginfo("Search sequence complete")

if __name__ == '__main__':
    try:
        node = SimpleSearch()
        rospy.spin()
    except rospy.ROSInterruptException:
        pass
