#!/usr/bin/env python3
import socket
import threading
import rospy
from std_msgs.msg import Bool

HOST = "0.0.0.0"
PORT = 5005


class RobotServer:
    def __init__(self):
        rospy.init_node("jcserver_mapping_ros")

        # 🔥 Publisher for autonomous trigger
        self.auto_pub = rospy.Publisher("/start_autonomous", Bool, queue_size=1)

        self.server_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self.server_socket.bind((HOST, PORT))
        self.server_socket.listen(5)

        print(f"[SERVER] Listening on {HOST}:{PORT}")

    def handle_client(self, client_socket, addr):
        print(f"[CONNECTED] {addr}")

        try:
            while True:
                data = client_socket.recv(1024).decode().strip()

                if not data:
                    break

                print(f"[RECV] {data}")

                if "<CMD:" in data:
                    cmd = data.replace("<CMD:", "").replace(">", "").strip()

                    # 🔥 NEW: Autonomous trigger
                    if cmd == "AUTO":
                        print("[AUTO] Triggering autonomous search")

                        # publish trigger
                        rospy.sleep(0.2)
                        self.auto_pub.publish(True)

                    else:
                        # Normal commands (F, A, D, S, K etc.)
                        self.handle_robot_command(cmd)

        except Exception as e:
            print(f"[ERROR] {e}")

        finally:
            client_socket.close()
            print(f"[DISCONNECTED] {addr}")

    def handle_robot_command(self, cmd):
        print(f"[CMD] {cmd}")

        # TODO: Replace this with your Arduino / Mega logic
        # Example:
        # serial.write(cmd.encode())

    def start(self):
        while not rospy.is_shutdown():
            client_socket, addr = self.server_socket.accept()
            client_thread = threading.Thread(
                target=self.handle_client,
                args=(client_socket, addr)
            )
            client_thread.start()


if __name__ == "__main__":
    server = RobotServer()
    server.start()
